    /**
     * Navigates to the Readings fragment in the dashboard activity
     */
    private void navigateToReadingsFragment() {
        FragmentActivity activity = getActivity();
        if (activity == null) {
            showErrorDialog("Unable to navigate: Host activity not found");
            return;
        }
        if (!(activity instanceof Dashboard)) {
            showErrorDialog("Unable to navigate: Invalid host activity");
            return;
        }

        Dashboard dashboard = (Dashboard) activity;
        Readings newReadingsFragment = new Readings();

        try {
            dashboard.getSupportFragmentManager()
                    .beginTransaction()
                    .replace(R.id.fragment_container, newReadingsFragment)
                    .setReorderingAllowed(true)
                    .commit();
            dashboard.getBottomNavigationView().setSelectedItemId(R.id.nav_home);
        } catch (Exception e) {
            e.printStackTrace();
            showErrorDialog("Navigation error: " + e.getMessage());
        }
    }

    /**
     * Shows a dialog confirming the transaction has been completed
     */
    private void showCompletionDialog() {
        new AlertDialog.Builder(getContext())
                .setTitle("Transaction Complete")
                .setMessage("Transaction has been successfully saved.")
                .setPositiveButton("OK", (dialog, which) -> {
                    Fragment parentFragment = getParentFragment();
                    if (parentFragment instanceof AccountValidationDialogFragment) {
                        AccountValidationDialogFragment dialogFragment = (AccountValidationDialogFragment) parentFragment;
                        Dialog parentDialog = dialogFragment.getDialog();
                        if (parentDialog != null && parentDialog.getWindow() != null) {
                            parentDialog.getWindow().setWindowAnimations(R.style.DialogAnimation);
                        }
                        dialogFragment.dismiss();
                    } else {
                        showErrorDialog("Unable to close dialog: Parent fragment not found");
                        return;
                    }

                    new android.os.Handler(android.os.Looper.getMainLooper()).postDelayed(() -> {
                        if (isAdded() && getActivity() != null) {
                            navigateToReadingsFragment();
                        }
                    }, 300);
                })
                .setNegativeButton("Cancel", (dialog, which) -> dialog.dismiss())
                .setCancelable(false)
                .show();
    }
